       ```rust
       /// Get anticipatory items formatted for heartbeat merge.
       /// Returns (items, is_first_heartbeat) tuple.
       pub(super) async fn get_anticipatory_for_heartbeat(&self) -> (Vec<AnticipatoryItem>, bool) {
           let runtime = self.anticipatory.read().await;
           let items = runtime.items.clone();
           let is_first = runtime.session_start_pending_at.is_some();
           (items, is_first)
       }
       ```

       ```rust
       // --- Phase 2.5: Gather anticipatory items for heartbeat merge (D-07, D-08, D-09) ---
       let (anticipatory_items, is_first_heartbeat) = self.get_anticipatory_for_heartbeat().await;

       let anticipatory_summary = if anticipatory_items.is_empty() {
           String::new()
       } else {
           anticipatory_items.iter()
               .map(|item| {
                   let priority_hint = if item.kind == "hydration" { "LOW-PRIORITY INFORMATIONAL" } else { "ACTIONABLE" };
                   format!("- [{}] {} (confidence: {:.2}): {} {}",
                       item.kind, item.title, item.confidence, item.summary,
                       if !item.bullets.is_empty() {
                           format!("\n    Bullets: {}", item.bullets.join(", "))
                       } else { String::new() }
                   )
               })
               .collect::<Vec<_>>()
               .join("\n")
       };
       ```
