```rust
pub struct PluginManifest {
    pub name: String,
    pub version: String,
    pub schema_version: u32,
    pub description: Option<String>,
    pub author: Option<String>,
    pub license: Option<String>,
    pub tamux_version: Option<String>,
    pub settings: Option<HashMap<String, SettingField>>,
    pub api: Option<ApiSection>,
    pub commands: Option<HashMap<String, CommandDef>>,
    pub skills: Option<Vec<String>>,
    pub auth: Option<AuthSection>,
    pub extra: HashMap<String, serde_json::Value>,
}

pub struct SettingField {
    pub field_type: String,       // "type" in JSON: "string", "number", "boolean", "select"
    pub label: String,
    pub required: bool,
    pub secret: bool,
    pub default: Option<serde_json::Value>,
    pub options: Option<Vec<String>>,
    pub description: Option<String>,
}

pub struct ApiSection {
    pub base_url: Option<String>,
    pub endpoints: HashMap<String, EndpointDef>,
    pub rate_limit: Option<RateLimitDef>,
}

pub struct EndpointDef {
    pub method: String,
    pub path: String,
    pub params: Option<HashMap<String, Value>>,
    pub headers: Option<HashMap<String, String>>,
    pub body: Option<HashMap<String, Value>>,
    pub response_template: Option<String>,
}

pub struct AuthSection {
    pub auth_type: String,          // "type" in JSON: "oauth2" or "api_key"
    pub authorization_url: Option<String>,
    pub token_url: Option<String>,
    pub scopes: Option<Vec<String>>,
    pub pkce: Option<bool>,
}

pub struct CommandDef {
    pub description: String,
    pub action: String,
}

pub struct RateLimitDef {
    pub requests_per_minute: u32,
}
```

Available Handlebars helpers (from template.rs):
- `{{urlencode value}}` - Percent-encode a string
- `{{json value}}` - Serialize to JSON string
- `{{default value "fallback"}}` - Return fallback if null/missing
- `{{truncate value N}}` - Truncate to N chars with "..."
- `{{join arr ", "}}` - Join array with separator

NOT available: `eq`, `if (eq ...)`, `math`, `dateFormat`, `lookup`
Strict mode: every variable referenced MUST exist in response JSON or use `default` helper.

```json
{
  "name": "gmail",
  "version": "1.0.0",
  "schema_version": 1,
  "description": "Read Gmail inbox and search emails",
  "author": "tamux",
  "license": "MIT",
  "tamux_version": ">=2.0.0",
  "auth": {
    "type": "oauth2",
    "authorization_url": "https://accounts.google.com/o/oauth2/v2/auth",
    "token_url": "https://oauth2.googleapis.com/token",
    "scopes": ["https://www.googleapis.com/auth/gmail.readonly"],
    "pkce": true
  },
  "settings": {
    "max_results": {
      "type": "number",
      "label": "Max Results",
      "required": false,
      "default": 10,
      "description": "Maximum number of messages to return"
    }
  },
  "api": {
    "base_url": "https://gmail.googleapis.com",
    "endpoints": {
      "list_inbox": {
        "method": "GET",
        "path": "/gmail/v1/users/me/messages?maxResults={{default params.max_results \"10\"}}&q={{urlencode (default params.query \"in:inbox\")}}",
        "response_template": "Found messages:\n\n{{#each messages}}Message ID: {{id}} (Thread: {{threadId}})\n{{/each}}\n\nTotal estimate: {{resultSizeEstimate}}"
      },
      "get_message": {
        "method": "GET",
        "path": "/gmail/v1/users/me/messages/{{params.message_id}}?format=METADATA&metadataHeaders=Subject&metadataHeaders=From&metadataHeaders=Date",
        "response_template": "Snippet: {{snippet}}\n\nHeaders:\n{{#each payload.headers}}- {{name}}: {{value}}\n{{/each}}\n\nLabels: {{join labelIds \", \"}}"
      },
      "search_messages": {
        "method": "GET",
        "path": "/gmail/v1/users/me/messages?maxResults={{default params.max_results \"10\"}}&q={{urlencode params.query}}",
        "response_template": "Search results:\n\n{{#each messages}}Message ID: {{id}} (Thread: {{threadId}})\n{{/each}}\n\nTotal estimate: {{resultSizeEstimate}}"
      }
    },
    "rate_limit": {
      "requests_per_minute": 60
    }
  },
  "commands": {
    "inbox": {
      "description": "Show recent inbox messages",
      "action": "list_inbox"
    },
    "search": {
      "description": "Search emails by query",
      "action": "search_messages"
    }
  },
  "skills": ["skills/gmail-inbox.yaml"]
}
```

Note on templates: Do NOT use `{{#if (eq ...)}}` -- the `eq` helper is not registered. Use `{{#each payload.headers}}` to iterate all headers and output them as `- name: value` lines. The agent's natural language capability extracts Subject/From/Date from the list. Use `{{default field "fallback"}}` for any optional field access. Do NOT include `"Authorization": "Bearer {{auth.access_token}}"` in endpoint headers -- the API proxy layer in Phase 17/18 automatically injects the OAuth token via the auth section; manual header injection would cause double-injection.

6. **Usage** -- `/gmail.inbox`, `/gmail.search`, `/calendar.today`, natural language queries
7. **Scopes** -- explain gmail.readonly is a restricted scope, Testing mode limits to 100 test users
8. **Creating Your Own Plugin** -- brief section pointing to plugin documentation (serves as template for community authors per D-10)
